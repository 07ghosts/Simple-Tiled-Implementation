Layer Callbacks

Saturday, May 31, 2014 1:32 AM

Every Layer has its own Update and Draw callback functions. When a map is initialized, each Layer is given an empty Update function and a Draw function respective of the Layer's type. When a Custom Layer is created, it is given empty callbacks. Both the Update and Draw callbacks can be overridden to customize your Map.

local sti = require("sti")

function love.load()
    map = sti.new("assets/maps/map01")

    -- Create a custom layer
    map:addCustomLayer("Sprite Layer", 2)

    -- Create a local shortcut to work with
    local spriteLayer = map.layers["Sprite Layer"]

    -- Add custom data
    spriteLayer.sprites = {}

    -- Create some data
    local sprite = {
        image = love.graphics.newImage("assets/sprites/player.png"),
        x = 128,
        y = 128,
        r = 0,
    }

    -- Insert the data into the layer
    table.insert(spriteLayer.sprites, sprite)

    -- Override update callback
    function spriteLayer:update(dt)
        -- Sprites will spin 90 degrees per second
        for _, sprite in ipairs(self.sprites) do
            sprite.r = self.sprite.r + math.rad(90 * dt)
        end
    end

    -- Override draw callback
    function spriteLayer:update(dt)
        -- Draw the sprites
        for _, sprite in ipairs(self.sprites) do
            love.graphics.draw(sprite.image, math.floor(sprite.x), math.floor(sprite.y), sprite.r)
        end
    end
end

function love.update(dt)
    map:update(dt)
end

function love.draw()
    map:draw()
end

function love.resize(w, h)
    map:resize(w, h)
end
